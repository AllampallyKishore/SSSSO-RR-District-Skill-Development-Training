Django Integration (Optional)
Create a Django project:

shell
django-admin startproject fileorganizer_web
cd fileorganizer_web
python manage.py startapp organizer
Add organizer app to INSTALLED_APPS in settings.py.

Create a file upload form in organizer/forms.py:

python
from django import forms

class UploadFileForm(forms.Form):
    file = forms.FileField()
View to handle uploads and invoke organization (in organizer/views.py):

python
from django.shortcuts import render
from .forms import UploadFileForm
import os
from .your_script import organize_files

def upload_and_organize(request):
    if request.method == "POST":
        form = UploadFileForm(request.POST, request.FILES)
        if form.is_valid():
            file = request.FILES['file']
            upload_path = "uploads/"
            os.makedirs(upload_path, exist_ok=True)
            with open(upload_path + file.name, 'wb+') as dest:
                for chunk in file.chunks():
                    dest.write(chunk)
            organize_files(upload_path)
            return render(request, "organizer/result.html")
    else:
        form = UploadFileForm()
    return render(request, "organizer/upload.html", {'form': form})
Wire up URLs and templates as required.
